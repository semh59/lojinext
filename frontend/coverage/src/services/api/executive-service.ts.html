export const executiveService = {
  getFvi: async (): Promise<FleetEfficiencyResponse> => {
    const r = await axiosInstance.get<FleetEfficiencyResponse>(
      "/reports/executive/kpi",
    );
    return r.data;
  },
 
  runWhatIf: async (payload: WhatIfRequest): Promise<WhatIfResponse> => {
    const r = await axiosInstance.post<WhatIfResponse>(
      "/reports/executive/what-if",
      payload,
    );
    return r.data;
  },
 
  getCarbon: async (days = 30): Promise<FleetCarbonResponse> => {
    const r = await axiosInstance.get<FleetCarbonResponse>(
      "/reports/executive/carbon",
      { params: { days } },
    );
    return r.data;
  },
 
  getCompliance: async (
    daysHorizon = 90,
  ): Promise<ComplianceHeatmapResponse> => {
    const r = await axiosInstance.get<ComplianceHeatmapResponse>(
      "/reports/executive/compliance",
      { params: { days_horizon: daysHorizon } },
    );
    return r.data;
  },
 
  getCashflow: async (days = 90): Promise<CashflowProjectionResponse> => {
    const r = await axiosInstance.get<CashflowProjectionResponse>(
      "/reports/executive/cashflow",
      { params: { days } },
    );
    return r.data;
  },
 
  getCrossFeature: async (days = 90): Promise<CrossFeatureImpactResponse> => {
    const r = await axiosInstance.get<CrossFeatureImpactResponse>(
      "/reports/executive/cross-feature",
      { params: { days } },
    );
    return r.data;
  },
 
  getBusFactor: async (n = 3): Promise<BusFactorResponse> => {
    const r = await axiosInstance.get<BusFactorResponse>(
      "/reports/executive/bus-factor",
      { params: { n } },
    );
    return r.data;
  },
 
  /** E.9 — CEO 1-pager PDF (blob download). */
  downloadPdf: async (): Promise<void> => {
    const r = await axiosInstance.get("/reports/executive/pdf", {
      responseType: "blob",
    });
    const blob = new Blob([r.data], { type: "application/pdf" });
    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");
    link.href = url;
    link.download = `executive-cockpit-${new Date()
      .toISOString()
      .slice(0, 10)}.pdf`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
  },
